1. 배열이란? ****중요****
- 같은 자료형(공간의 크기가 동일)
- 물리적인 나열 구조(순차 자료 구조)


	// 향상된 for문 : 배열 쉽게 반복
	
	package exam02;
	public class Ex04 {
		public static void main(String[] args) {
			String[] str = {"가나", "다라", "마바"};
			
			for(int i = 0; i < str.length; i++) { // 일반 for문
				System.out.println(str[i]);
			}
			for(String s: str) {         // 향상된 for문
				System.out.println(s); 
			}
		}
	}
